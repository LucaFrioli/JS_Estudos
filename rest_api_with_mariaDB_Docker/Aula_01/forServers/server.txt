# passo 1 : REMOVE VERSÃ•ES ANTERIORES DO DOCKER
sudo apt-get remove \
    docker \
    docker-engine \
    docker.io \
    containerd runc -y

# passo 2 :
sudo apt update


# passo 3 : INSTALA O DOCKER-CE
sudo apt install \
    apt-transport-https \
    ca-certificates \
    curl \
    gnupg-agent \
    software-properties-common -y

# passo 4 :
curl -fsSL https://download.docker.com/linux/ubuntu/gpg |

# passo 5
sudo apt-key add -

# passo 6 :
sudo add-apt-repository \
   "deb [arch=amd64] https://download.docker.com/linux/ubuntu \
   $(lsb_release -cs) \
   stable" -y

# passo 7 :
sudo apt update

# passo 8 :
sudo apt install docker-ce docker-ce-cli containerd.io -y
